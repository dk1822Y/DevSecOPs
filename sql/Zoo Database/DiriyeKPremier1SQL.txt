Question 5

Question 6

FROM PART

Question 7

SELECT PART_NUM, DESCRIPTION, ON_HAND
FROM PART
WHERE ON_HAND BETWEEN 10 AND 25;

FROM PART
WHERE ON_HAND >= 10
  AND ON_HAND <= 25;

Question 9

Question 14

FROM CUSTOMER

Question 15

SELECT CUSTOMER_NUM, CUSTOMER_NAME, CREDIT_LIMIT, BALANCE AS TOTAL_BALANCE
FROM CUSTOMER
WHERE REP_NUM = 65 
  AND BALANCE < CREDIT_LIMIT;

Question 16

SELECT PART_NUM, DESCRIPTION, (ON_HAND * PRICE) AS ON_HAND_VALUE
FROM PART
WHERE ON_HAND > (SELECT AVG(ON_HAND) FROM PART);

Question 17

SELECT MIN(PRICE) AS LOWEST_PRICE
FROM PART;

Question 18

SELECT PART_NUM, DESCRIPTION, PRICE 
FROM PART
WHERE PRICE = (SELECT MAX(PRICE) FROM PART);

Question 19

SELECT REP_NUM, SUM(BALANCE) AS BALANCE_TOTAL
FROM CUSTOMER
GROUP BY REP_NUM
ORDER BY REP_NUM;

Question 20

SELECT REP_NUM, SUM(BALANCE) AS BALANCE_TOTAL
FROM CUSTOMER
HAVING SUM(BALANCE) > 10000
GROUP BY REP_NUM
ORDER BY REP_NUM;

Question 21
