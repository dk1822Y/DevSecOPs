Question 6

SELECT P.PRODUCT_NAME, PP.PRODUCT_PURCHASE_DATE, MP.PRODUCT_FINISH_DATE
FROM PRODUCT P LEFT JOIN MANUFACTURED_PRODUCT MP ON P.PRODUCT_ID = MP.PRODUCT_ID 
JOIN PURCHASED_PRODUCT PP ON P.PRODUCT_ID = PP.PRODUCT_ID
ORDER BY PP.PRODUCT_PURCHASE_DATE, MP.PRODUCT_FINISH_DATE;

Question 8

Question 9

SELECT FC.FOOD_CATEGORY_DESCRIPTION, FI.FOOD_ITEM_DESCRIPTION, FI.FOOD_QOH
FROM FOOD_CATEGORY FC JOIN FOOD_ITEM FI USING (FOOD_CATEGORY_ID)
WHERE FI.FOOD_ITEM_DESCRIPTION LIKE 'Wholesome%' 
OR FC.FOOD_CATEGORY_DESCRIPTION LIKE 'A%';

Question 10

SELECT FOOD_ITEM_DESCRIPTION, FOOD_CATEGORY_DESCRIPTION
FROM FOOD_ITEM
FULL JOIN FOOD_CATEGORY ON FOOD_ITEM.FOOD_CATEGORY_ID = FOOD_CATEGORY.FOOD_CATEGORY_ID
WHERE FOOD_ITEM_DESCRIPTION LIKE 'Tasty%' 
OR FOOD_ITEM_DESCRIPTION LIKE 'Organic%' 
OR FOOD_CATEGORY_DESCRIPTION LIKE 'A%'
OR FOOD_CATEGORY_DESCRIPTION LIKE 'B%';

Question 12

Question 13

Question 14

Question 15

Question 16

Question 19

SELECT ANIM_NAME, ANIM_BREED_NAME
FROM ANIMAL A
JOIN ANIMAL_CHECKUP AC ON A.ANIM_ID = AC.ANIM_ID
JOIN ANIMAL_BREED AB ON A.ANIM_BREED_ID = AB.ANIM_BREED_ID
WHERE AC.ANIM_CHECKUP_GENERAL_HEALTH LIKE '%Malnourished%'
AND ROWNUM <= 30;

Question 20

Question 21

SELECT FOOD_ITEM_DESCRIPTION, FOOD_QOH, FS.FOOD_SUP_COMPANY_NAME
FROM FOOD_ITEM FI
JOIN FOOD_ITEM_SUPPLIER FIS
JOIN FOOD_SUPPLIER FS
ON FIS.FOOD_SUP_ID = FS.FOOD_SUP_ID
ON FI.FOOD_ITEM_ID = FIS.FOOD_ITEM_ID

Question 22

SELECT VOL_FIRST_NAME, VOL_LAST_NAME, VOL_START_DATE, ZOO_NAME, DEPT_NAME
FROM DEPARTMENT JOIN ZOO ON ZOO.ZOO_ID = DEPARTMENT.ZOO_ID
JOIN VOLUNTEER_DEPT_HISTORY ON VOLUNTEER_DEPT_HISTORY.DEPT_ID = DEPARTMENT.DEPT_ID
JOIN VOLUNTEER ON VOLUNTEER.VOL_ID = VOLUNTEER_DEPT_HISTORY.VOL_ID
WHERE VOL_START_DATE LIKE '%-MAR-10%';

Question 23 

SELECT S.STORE_ADDRESS, S.STORE_CITY, E.EMP_FIRST_NAME, E.EMP_LAST_NAME, T.TRANS_TOTAL_AMT
FROM STORE S, EMPLOYEE E, TRANSACTION T
WHERE S.STORE_ID = T.STORE_ID
AND T.EMP_ID = E.EMP_ID
AND T.TRANS_TOTAL_AMT IN (SELECT MAX(TRANS_TOTAL_AMT) FROM TRANSACTION);

Question 24

SELECT A.ANIM_NAME, AB.ANIM_BREED_NAME, ALE.ANIM_LIVING_ENVIRONMENT_TYPE, I.ILL_TYPE_ID
FROM ANIMAL A 
JOIN ANIMAL_BREED AB ON A.ANIM_BREED_ID = AB.ANIM_BREED_ID
JOIN ANIMAL_LIVING_ENVIRONMENT ALE ON ALE.ANIM_LIVING_ENVIRONMENT_ID = A.ANIM_LIVING_ENVIRONMENT_ID 
JOIN ILLNESS I ON I.ANIM_ID = A.ANIM_ID
JOIN PRESCRIPTION P ON A.ANIM_ID = P.ANIM_ID
JOIN VET V ON V.VET_ID = P.VET_ID
WHERE V.VET_FIRST_NAME = 'Debra'
AND ANIM_LIVING_ENVIRONMENT_TYPE = 'Desert';
